Table users {
  id integer [primary key]
  username varchar
  email varchar
  hashed_password varchar
}

Table trips {
  id integer [primary key]
  startAddress varchar
  startLat float
  startLng float
  endAddress varchar
  endLat float
  endLng float
  duration int
  distance varchar
  userId int
  created_at timestamp
  modified_at timestamp
}

Table detours {
  id integer [primary key]
  name varchar
  description text
  address varchar
  lat float
  long float
  tripId int
  originWaypoint int
  stopNumber int
  created_at timestamp
  modified_at timestamp
}

Table waypoints {
  id integer [primary key]
  tripId int
  lat float
  long float
  time int
  created_at timestamp
  modified_at timestamp
}

Table preference_tags {
  id integer [primary key]
  userId int
  tag varchar
}

Ref: trips.userId > users.id 

Ref: detours.tripId > trips.id

Ref: waypoints.tripId > trips.id

Ref: preference_tags.userId > users.id

Ref: detours.originWaypoint > waypoints.id
